\subsection*{2 Big Oh notation}


\begin{frame}[fragile]{largest element}
\begin{lstlisting}[numbers=left]
int search(int a[],int n)
{
   int i;
   int best_val=a[0];

   for(i=1;i<n;i++){
      if(a[i]>best_val)
         best_val= a[i];
   }

  return best_val;
}
\end{lstlisting}
