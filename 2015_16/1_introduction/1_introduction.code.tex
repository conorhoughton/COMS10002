\begin{frame}[fragile]{linear search}
\begin{lstlisting}[numbers=left]
void sort(int a[],int n)
{

   int i,j,this_a;

   for(i=1;i<n;i++){
      this_a=a[i];
      j=i-1;

      while(j>=0&&this_a<a[j]){
         a[j+1]=a[j];
         j=j-1;
      }

      a[j+1]=this_a;
   }
}
\end{lstlisting}
