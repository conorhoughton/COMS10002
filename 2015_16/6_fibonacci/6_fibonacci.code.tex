\subsection*{II.2 Fibonacci}
\begin{frame}[fragile]{Fibonacci}
\begin{lstlisting}[numbers=left]
int fib(int n)
{
  if(n==0||n==1){
      return n;
  }

  return fib(n-1)+fib(n-2);

}
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Fibonacci with ternary operator}
\begin{lstlisting}[numbers=left]
int fib(int n)
{
  return (n < 2) & n : fib(n-1) + fib(n-2); 
}
\end{lstlisting}
\end{frame}



\begin{frame}[fragile]{Fibonacci with tail recursion}
\begin{lstlisting}[numbers=left]
int fib_r(int n, int a, int b)
{
   if(n==0)
     return a;
   if(n==1)
     return b;
   if(n==2)
     return a+b;

   return fib_r(n-1,b,a+b);
}

int fib(int n)
{
  return fib_r(n,0,1);
}
\end{lstlisting}
\end{frame}



\begin{frame}[fragile]{Fibonacci using a loop}
\begin{lstlisting}[numbers=left]
int fib(int n)
{
  if(n<2)
     return n;
  
  int last=1, old_last=0;
  int i;

  for(i=2;i<=n;i++){
        int temp=last;
        last=last+old_last;
        old_last=temp;
  }
  return last;
}
\end{lstlisting}
\end{frame}
